    <div>
        <label for="date">Date (MM/DD/YYYY):</label>
        <input type="text" id="date" placeholder="12/25/2024">
    </div>
    <div>
        <label for="person">Your Number (1-20):</label>
        <input type="number" id="person" placeholder="1">
    </div>
    <button onclick="calculateKaphisma()">Get My Kaphisma</button>
    <div id="result"></div>

    <script>
      function calculateKaphisma() {
        const dateInput = document.getElementById("date").value;
        const personInput = parseInt(document.getElementById("person").value);
        const resultDiv = document.getElementById("result");

        if (!dateInput || isNaN(personInput) || personInput < 1 || personInput > 20) {
            resultDiv.textContent = "Please enter a valid date (MM/DD/YYYY) and a person number (1-20)";
            return;
        }

        const startDate = new Date("12/18/2024");
        const inputDate = new Date(dateInput);

        const days = Math.floor((inputDate - startDate) / (1000 * 60 * 60 * 24));

       if (isNaN(days)) {
           resultDiv.textContent = "Please enter a valid date (MM/DD/YYYY)";
           return;
          }

        const kaphisma = ((days + personInput - 1) % 20) + 1;
        resultDiv.textContent = kaphisma;
    }
    </script>
